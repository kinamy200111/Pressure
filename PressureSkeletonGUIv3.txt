-- ===== ЭФФЕКТ CIRCLE5: RIPPLE BURST каждые 10 сек =====
spawn(function()
    local IDLE_PERIOD = 10
    local BASE_RADIUS = 5
    local BASE_COLOR  = Color3.fromHex("#ad47fe")
    local lastBurst   = os.clock() - IDLE_PERIOD

    while true do
        wait(0.016)
        if not Circle5.Visible then continue end
        local now = os.clock()
        local timeSinceBurst = now - lastBurst

        local breath = math.sin(now * 1.2) * 0.8
        Circle5.Radius        = BASE_RADIUS + breath
        Circle5_glow1.Transparency = 0.75 + math.sin(now * 1.2) * 0.15
        Circle5_glow1.Radius  = 9 + breath
        Circle5_glow2.Transparency = 0.88
        Circle5_glow2.Radius  = 13
        Circle5_glow3.Transparency = 0.93
        Circle5_glow3.Radius  = 17

        if timeSinceBurst >= IDLE_PERIOD then
            lastBurst = now
            local flashDur   = 0.25
            local flashStart = os.clock()
            while os.clock() - flashStart < flashDur do
                wait(0.016)
                if not Circle5.Visible then break end
                local p  = (os.clock() - flashStart) / flashDur
                local r  = BASE_RADIUS + math.sin(p * math.pi) * 4
                Circle5.Radius = r
                local lp = math.sin(p * math.pi)
                Circle5.Color  = Color3.new(
                    0xad/255 + (1 - 0xad/255) * lp,
                    0x47/255 + (1 - 0x47/255) * lp,
                    0xfe/255 + (1 - 0xfe/255) * lp
                )
                Circle5_glow1.Radius = 10 + lp * 8  Circle5_glow1.Transparency = 0.3  - lp * 0.25
                Circle5_glow2.Radius = 15 + lp * 10 Circle5_glow2.Transparency = 0.6  - lp * 0.4
                Circle5_glow3.Radius = 20 + lp * 14 Circle5_glow3.Transparency = 0.75 - lp * 0.5
            end
            Circle5.Color  = BASE_COLOR
            Circle5.Radius = BASE_RADIUS

            local rippleDur = 0.7
            local delays    = {0, 0.12, 0.24}
            for _, delay in ipairs(delays) do
                spawn(function()
                    wait(delay)
                    if not Circle5.Visible then return end
                    local rStart = os.clock()
                    while os.clock() - rStart < rippleDur do
                        wait(0.016)
                        if not Circle5.Visible then break end
                        local rp    = (os.clock() - rStart) / rippleDur
                        local ringR = 6 + rp * 22
                        Circle5_glow1.Position = Circle5.Position
                        Circle5_glow2.Position = Circle5.Position
                        Circle5_glow3.Position = Circle5.Position
                        if delay == 0 then
                            Circle5_glow1.Radius = ringR Circle5_glow1.Transparency = 0.2 + rp * 0.8
                        elseif delay == 0.12 then
                            Circle5_glow2.Radius = ringR Circle5_glow2.Transparency = 0.3 + rp * 0.7
                        else
                            Circle5_glow3.Radius = ringR Circle5_glow3.Transparency = 0.4 + rp * 0.6
                        end
                    end
                    if delay == 0 then
                        Circle5_glow1.Radius = 9  Circle5_glow1.Transparency = 0.75
                    elseif delay == 0.12 then
                        Circle5_glow2.Radius = 13 Circle5_glow2.Transparency = 0.88
                    else
                        Circle5_glow3.Radius = 17 Circle5_glow3.Transparency = 0.93
                    end
                end)
            end
        end
    end
end)

-- ===== ЭФФЕКТ ТЕКСТА RATHUB: слабое свечение + slow color breathe =====
-- Прозрачность glow-слоёв тихо колышется через sin с малой амплитудой
-- Раз в ~8 сек текст делает короткий "shimmer" — цвет проходит белый и возвращается
spawn(function()
    local SHIMMER_PERIOD = 8
    local lastShimmer    = os.clock() - SHIMMER_PERIOD

    -- базовые офсеты glow-слоёв от Main1.Position + (54,17)
    local offsets = {
        {obj=TextRATHUB_g1, dx=-1, dy=-1},
        {obj=TextRATHUB_g2, dx= 1, dy= 1},
        {obj=TextRATHUB_g3, dx=-2, dy=-2},
        {obj=TextRATHUB_g4, dx= 2, dy= 2},
    }
    -- базовые прозрачности
    local baseTrans = {0.82, 0.82, 0.88, 0.88}

    while true do
        wait(0.016)
        if not TextRATHUB.Visible then continue end

        local now  = os.clock()
        local wave = math.sin(now * 0.9)  -- медленная волна -1..1

        -- idle: glow чуть дышит (амплитуда ±0.06)
        for i, g in ipairs(offsets) do
            g.obj.Transparency = baseTrans[i] + wave * 0.06
        end

        -- shimmer каждые SHIMMER_PERIOD сек
        if now - lastShimmer >= SHIMMER_PERIOD then
            lastShimmer = now
            local shimDur   = 0.5
            local shimStart = os.clock()
            while os.clock() - shimStart < shimDur do
                wait(0.016)
                if not TextRATHUB.Visible then break end
                local p  = (os.clock() - shimStart) / shimDur  -- 0..1
                -- арка: sin(p*pi) = 0→1→0
                local lp = math.sin(p * math.pi)

                -- основной текст: белеет чуть-чуть
                TextRATHUB.Color = Color3.new(
                    1,
                    1 - lp * 0.15,  -- слегка теряет G и B → лёгкий тёплый оттенок
                    1 - lp * 0.05
                )

                -- glow-слои: ярче и чуть крупнее смещение во время shimmer
                for i, g in ipairs(offsets) do
                    g.obj.Transparency = baseTrans[i] - lp * 0.35
                    g.obj.Color = Color3.new(
                        0x9d/255 + lp * (1 - 0x9d/255),
                        0x3d/255 + lp * (1 - 0x3d/255),
                        0xf0/255 + lp * (1 - 0xf0/255)
                    )
                end
            end
            -- сброс
            TextRATHUB.Color = Color3.fromHex("#FFFFFF")
            for i, g in ipairs(offsets) do
                if i <= 2 then g.obj.Color = Color3.fromHex("#9d3df0")
                else            g.obj.Color = Color3.fromHex("#6b1db0") end
                g.obj.Transparency = baseTrans[i]
            end
        end
    end
end)

-- ===== DRAG + КЛИКИ =====
spawn(function()
    local dragging = false
    local dragStart, startPos
    local lastMouse1 = false
    local lastF1     = false
    local guiVisible = true

    while true do
        wait(0.01)
        if not isrbxactive() then lastMouse1=false lastF1=false continue end

        local mouse1 = ismouse1pressed()
        local mPos   = Vector2.new(Mouse.X, Mouse.Y)
        local f1     = iskeypressed(0x70)

        if f1 and not lastF1 then
            guiVisible = not guiVisible
            for _,o in ipairs(allObjs) do o.Visible = guiVisible end
            for _,sq in ipairs(VeryTopPlace_Parts) do sq.Visible = guiVisible end
            if not guiVisible then
                ContentPageVisuals.Visible   = false ContentPageVisuals_Border.Visible   = false
                ContentPageExploits.Visible  = false ContentPageExploits_Border.Visible  = false
                ContentPageMisc.Visible      = false ContentPageMisc_Border.Visible      = false
                ContentPageSettings.Visible  = false ContentPageSettings_Border.Visible  = false
            end
        end

        if mouse1 and not lastMouse1 and guiVisible then
            local tp = VeryTopPlace_Parts[1]
            if mPos.X >= tp.Position.X and mPos.X <= tp.Position.X+604 and
               mPos.Y >= tp.Position.Y and mPos.Y <= tp.Position.Y+46 then
                dragging  = true
                dragStart = mPos
                startPos  = Main1.Position
            end
            for i,td in ipairs(tabPages) do
                local t = td.tab
                if mPos.X >= t.Position.X and mPos.X <= t.Position.X+t.Size.X and
                   mPos.Y >= t.Position.Y and mPos.Y <= t.Position.Y+t.Size.Y then
                    switchTab(i)
                end
            end
        end

        if not mouse1 then dragging = false end

        if dragging and mouse1 then
            local np  = startPos + (mPos - dragStart)
            Main1.Position = np
            local gW = 604
            for i,sq in ipairs(VeryTopPlace_Parts) do
                sq.Position = Vector2.new(np.X+(i-1)*math.floor(gW/50), np.Y)
            end
            -- Circle5
            local c5p = np + Vector2.new(25,23)
            Circle5.Position       = c5p
            Circle5_glow1.Position = c5p
            Circle5_glow2.Position = c5p
            Circle5_glow3.Position = c5p
            -- TextRATHUB glow слои
            local tp = np + Vector2.new(54,17)
            TextRATHUB.Position    = tp
            TextRATHUB_g1.Position = np + Vector2.new(53,16)
